```dart
// 根据月份自动判断季节
String getCurrentSeason() {
  final month = DateTime.now().month;
  if (month >= 3 && month <= 5) return '春';
  if (month >= 6 && month <= 8) return '夏';
  if (month >= 9 && month <= 11) return '秋';
  return '冬';
}
```
